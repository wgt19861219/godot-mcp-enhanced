const fs = require('fs');

const setMatch = validationSrc.match(/KNOWN_BASE_METHODS\s*=\s*new\s+Set<string>\(\[([\s\S]*?)\]\)/);

## Task 3: WebSocket 消息大小/连接数限制

**Files:**
- Modify: `addons/godot_mcp_server/websocket_server.gd`
- Modify: `src/core/EditorConnection.ts`

- [ ] **Step 1: 添加常量到 websocket_server.gd**

在文件顶部常量区域添加：
```gdscript
const MAX_PEERS := 5
const MAX_MESSAGE_SIZE := 1048576  # 1MB
```

- [ ] **Step 2: 修改连接接受逻辑**

在 `_process` 函数中，`_server.is_connection_available()` 分支内，修改连接接受逻辑：

将：
```gdscript
if _server.is_connection_available():
    var tcp_peer = _server.take_connection()
    var ws_peer = WebSocketPeer.new()
    ws_peer.accept_stream(tcp_peer)
    _peers.append(ws_peer)
    print("[MCP] Client connected (total: %d)" % _peers.size())
    _update_panel("MCP: %d client(s) connected" % _peers.size())
```

改为：
```gdscript
if _server.is_connection_available():
    var tcp_peer = _server.take_connection()

    if _peers.size() >= MAX_PEERS:
        tcp_peer.disconnect()
        push_warning("[MCP] Connection rejected: max peers reached (%d)" % MAX_PEERS)
        return

    var ws_peer = WebSocketPeer.new()
    ws_peer.set_inbound_buffer_size(MAX_MESSAGE_SIZE)
    ws_peer.accept_stream(tcp_peer)
    _peers.append(ws_peer)
    print("[MCP] Client connected (total: %d)" % _peers.size())
    _update_panel("MCP: %d client(s) connected" % _peers.size())
```

- [ ] **Step 3: 修改 EditorConnection.ts 消息处理**

在 `src/core/EditorConnection.ts` 的 WebSocket message handler 中添加消息大小检查。

文件顶部添加常量：
```typescript
const MAX_INBOUND_MESSAGE_SIZE = 1048576; // 1MB
```

在 message handler 中，JSON.parse 之前添加检查：
```typescript
this.ws.on('message', (data: WebSocket.Data) => {
  const raw = typeof data === 'string' ? data : data.toString();
  if (raw.length > MAX_INBOUND_MESSAGE_SIZE) {
    console.warn('[MCP Editor] Inbound message exceeds size limit, discarding');
    return;
  }
  // ... existing JSON.parse logic
});
```
